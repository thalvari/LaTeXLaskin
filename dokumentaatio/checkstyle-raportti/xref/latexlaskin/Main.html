6   package latexlaskin;
7   
8   import java.util.List;
9   import latexlaskin.calculator.Calculator;
10  
11  /**
12   *
13   * @author thalvari
14   */
15  public class Main {
16  
17      private static final String APPID = "WJ628E-G3H5VTERP4";
18      private static final boolean DEBUG = true;
19      private static final String FORMAT = "plaintext";
20      private static final String INPUT = "|2 x - 2|";
21  
22      private static Calculator calc;
23  
24      public static void main(String[] args) {
25          if (args.length > 1) {
26              System.out.println("Anna parametrina korkeintaan yksi merkkijono.");
27              return;
28          }
29  
30          String input = INPUT;
31          if (args.length == 1) {
32              input = args[0];
33          }
34  
35          calc = new Calculator(APPID, DEBUG, FORMAT);
36          List<String> results = calc.query(input);
37          printResults(results);
38      }
39  
40      private static void printResults(List<String> results) {
41          if (results == null) {
42              System.out.println(calc.getError());
43              return;
44          }
45  
46          System.out.println("Ratkaisut LaTeX-koodina:");
47          for (String result : results) {
48              System.out.println(result);
49          }
50      }
51  }
