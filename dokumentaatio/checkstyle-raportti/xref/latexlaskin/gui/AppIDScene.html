6   package latexlaskin.gui;
7   
8   import java.io.File;
9   import javafx.event.ActionEvent;
10  import javafx.scene.control.Button;
11  import javafx.scene.control.Label;
12  import javafx.scene.control.PasswordField;
13  import javafx.stage.Stage;
14  import latexlaskin.calculator.Calculator;
15  import latexlaskin.io.IO;
16  
17  /**
18   *
19   * @author thalvari
20   */
21  public class AppIDScene extends MyScene {
22  
23      private PasswordField appIDPasswordField;
24      private Button replaceButton;
25      private Button clearButton;
26      private Button returnButton;
27      private Label errorLabel;
28  
29      public AppIDScene(Stage stage, Calculator calc) {
30          super(stage, calc);
31      }
32  
33      @Override
34      protected void construct() {
35          appIDPasswordField = new PasswordField();
36          appIDPasswordField.setPromptText("AppID");
37          replaceButton = new Button("Vaihda");
38          replaceButton.setOnAction(this::replaceButtonHandle);
39          clearButton = new Button("Tyhjenn√§");
40          clearButton.setOnAction(this::appIDButtonHandle);
41          returnButton = new Button("Palaa");
42          returnButton.setOnAction(this::returnButtonHandle);
43          errorLabel = new Label();
44          gridPane.add(appIDPasswordField, 0, 0);
45          gridPane.add(replaceButton, 1, 0);
46          gridPane.add(clearButton, 2, 0);
47          gridPane.add(returnButton, 3, 0);
48          gridPane.add(errorLabel, 0, 1, 3, 1);
49      }
50  
51      private void appIDButtonHandle(ActionEvent event) {
52          appIDPasswordField.clear();
53      }
54  
55      private void replaceButtonHandle(ActionEvent event) {
56          String oldAppID = calc.getAppID();
57          String newAppID = appIDPasswordField.getText();
58          calc.setAppID(newAppID);
59          calc.query("1 + 1");
60          if (calc.getError() == null) {
61              IO.writeAppID(new File("APPID"), newAppID);
62              appIDPasswordField.clear();
63              errorLabel.setText("");
64              stage.setScene(otherScene.getScene());
65          } else {
66              calc.setAppID(oldAppID);
67              errorLabel.setText(calc.getError());
68          }
69          stage.sizeToScene();
70      }
71  
72      private void returnButtonHandle(ActionEvent event) {
73          appIDPasswordField.clear();
74          errorLabel.setText("");
75          stage.setScene(otherScene.getScene());
76      }
77  }
