6   package latexlaskin.gui;
7   
8   import java.util.List;
9   import javafx.geometry.Insets;
10  import javafx.scene.control.Button;
11  import javafx.scene.control.Label;
12  import javafx.scene.control.TextField;
13  import javafx.stage.Stage;
14  import latexlaskin.calculator.Calculator;
15  
16  /**
17   *
18   * @author thalvari
19   */
20  public class CalcScene extends MyScene {
21  
22      public CalcScene(Stage stage, Calculator calc) {
23          super(stage, calc);
24      }
25  
26      @Override
27      protected void construct() {
28          Label inputLabel = new Label("Sy√∂te: ");
29          TextField inputField = new TextField();
30          Label resultLabel = new Label();
31          Button calcButton = new Button("Laske");
32          gridPane.add(inputLabel, 0, 1);
33          gridPane.add(inputField, 1, 1);
34          gridPane.add(resultLabel, 1, 3);
35          gridPane.add(calcButton, 1, 2);
36          gridPane.setHgap(10);
37          gridPane.setVgap(10);
38          gridPane.setPadding(new Insets(15, 15, 15, 15));
39          calcButton.setOnAction((event) -> {
40              List<String> results = calc.query(inputField.getText());
41              if (results == null) {
42                  resultLabel.setText(calc.getError());
43                  stage.sizeToScene();
44                  return;
45              }
46  
47              resultLabel.setText("Ratkaisut:");
48              int i = 4;
49              for (String result : results) {
50                  TextField textField = new TextField(result);
51                  textField.setEditable(false);
52                  gridPane.add(textField, 1, i);
53                  i++;
54              }
55              stage.sizeToScene();
56          });
57      }
58  }
