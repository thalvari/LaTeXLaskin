6   package latexlaskin.gui;
7   
8   import javafx.geometry.Insets;
9   import javafx.scene.control.Button;
10  import javafx.scene.control.Label;
11  import javafx.scene.control.TextField;
12  import javafx.stage.Stage;
13  import latexlaskin.calculator.Calculator;
14  
15  /**
16   *
17   * @author thalvari
18   */
19  public class LoginScene extends MyScene {
20  
21      public LoginScene(Stage stage, Calculator calc) {
22          super(stage, calc);
23      }
24  
25      @Override
26      protected void construct() {
27          Label label = new Label("AppID: ");
28          TextField textField = new TextField();
29          Button button = new Button("Kirjaudu");
30          Label errorLabel = new Label();
31          gridPane.add(label, 0, 0);
32          gridPane.add(textField, 1, 0);
33          gridPane.add(button, 1, 1);
34          gridPane.add(errorLabel, 1, 2);
35          gridPane.setHgap(10);
36          gridPane.setVgap(10);
37          gridPane.setPadding(new Insets(15, 15, 15, 15));
38          button.setOnAction((event) -> {
39              calc.setAppID(textField.getText());
40              calc.query("1 + 1");
41              if (calc.getError() == null) {
42                  stage.setScene(otherScene.getScene());
43              } else {
44                  errorLabel.setText(calc.getError());
45              }
46          });
47      }
48  }
