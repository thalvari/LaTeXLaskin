10  public interface WASourceInfo extends Visitable {
11      
12      // There are two forms for the HTML data for source info. The first is a link to "raw" HTML that would be easy for clients to operate
13      // on to format as desired. This is the URL that Android and iPhone request, although server-side code detects this
14      // and replaces the file with a specially-formatted one. This wouldn't happen for other API clients. This URL is what is
15      // returned by getURL(). The second form for source info is the HTML page that you would see on the website
16      // (it has a banner that looks like the rest of the website). Clients might want to direct users to that page instead,
17      // and this is what is returned by getFormattedURL().
18      String getURL();
19      String getFormattedURL();
20      String getText();
