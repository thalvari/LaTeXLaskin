21  // This class is written exactly like WAImageImpl, as if sound downloads will be done in advance and stored
22  // as files to be passed to the media player. The player also supports streaming, though, and that would be
23  // a better choice. At the moment, there appear to be problems with streaming. I have encountered failures in MEdiaPlayer.prepare(),
24  // and although these might be problems with our servers not being set up to deliver progressive downloads,
25  // the internet is filled with people claiming problems with streaming on various devices and Android versions.
26  // Even if we switch to a streaming mode, it makes sense for this class to support download-to-file mode. We can
27  // avoid using the feature if we decide streaming is working.
28  
29  public class WASoundImpl implements WASound, Visitable, Serializable {
30  
31      private String url;
32      private String format;
33      
34      private volatile File file;
35      private volatile int cachedHashCode = 0;
36      private volatile boolean soundAcquired = false;
37      
38      private transient HttpProvider http;
39      private transient File tempDir;
40     
41      static final WASoundImpl[] EMPTY_ARRAY = new WASoundImpl[0];
42  
43      private static final long serialVersionUID = 3863860206159745210L;
44  
45      
46      WASoundImpl(Element thisElement, HttpProvider http, File tempDir) throws WAException {
47          
48          url = thisElement.getAttribute("url");
49          format = thisElement.getAttribute("type");
50          if (format.equals("audio/x-wav"))
51              format = "WAV";
52          else if (format.equals("audio/midi"))
53              format = "MIDI";
54          this.http = http;
55          this.tempDir = tempDir;
56      }
57  
58      
59      ////////////////////  WASound interface  //////////////////////////////
60      
61      public String getURL() {
62          return url;
63      }
64      
65      public String getFormat() {
66          return format;
67      }
68      
69      public synchronized File getFile() {
70          return file;
71      }
72      
73      public void acquireSound() {
74          
75          // If this is a deserialized instance, http will  be null. Such instances are "dead"; they can
76          // never retrieve new content from the web.
77          // The only synchronization needed here is that soundAcquired is volatile.
78          if (!soundAcquired && http != null) {
79              if (!url.equals("")) {
80                  try {
81                      String suffix;
82                      if (format == "WAV")
83                          suffix = ".wav";
84                      else if (format == "MIDI")
85                          suffix = ".mid";
86                      else 
87                          suffix = ".tmp";
88                      String outFile = File.createTempFile("WASound", suffix, tempDir).getAbsolutePath();
89                      URLFetcher fetcher = new URLFetcher(new URL(url), outFile, http, null);
90                      fetcher.fetch();
91                      setFile(fetcher.getFile());
92                  } catch (Exception e) {
93                      // TODO: report?
94                  }
95                  
96              }
97              soundAcquired = true;
98          }
99      }
100

102     ////////////////////////  hashCode()  /////////////////////////
103     
104     // We use hashCode() as a "content code" to tell us quickly whether the object's content
105     // has changed since some point in the past. Note that we do not override equals() as well, 
106     // but it is not necessary to override equals() when overriding hashCode() (although it _is_
107     // necessary to override hashCode() when overriding equals()). Every field on which the
108     // hashcode depends must be either immutable or volatile (or otherwise controlled by synchronized
109     // blocks), as these values are changed on a background thread.
110     // This is not a particularly good hash function, but it doesn't need to be. The only property
111     // that really matters is that its value changes when the content of this object changes.
112     
113     public synchronized int hashCode() {
114         
115         if (cachedHashCode != 0)
116             return cachedHashCode;
117         
118         int result = 17;
119         result = 37*result + url.hashCode();
120         if (file != null)
121             result = 37*result + file.hashCode();
122         cachedHashCode = result;
123         return result;
124     }
125     
126     
127     ////////////////////////////////////////////
128     
129     private synchronized void setFile(File file) {
130         this.file = file;
131         cachedHashCode = 0;  // Force recompute of hash now that content has changed.
132     }
133     
134     ///////////////////////////  Visitor interface  ////////////////////////////
135     
136     public void accept(Visitor v) {
137         v.visit(this);
138     }

140 }
