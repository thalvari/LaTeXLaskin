19  /**
20   * Responsible for all things related to proxies. This includes storing the manually-set proxy
21   * host and port (if any), deciding what proxy, if any, to use for a given request
22   * (determined by manual or system settings), and serving up username/password credentials if required
23   * for an authenticating proxy.
24   *
25   * The one thing this class does not do is handle prompting the user for username and password
26   * if necessary. That is handled in Mathematica code triggered by a 407 HTTP status.
27   *
28   * The HttpHandler instance holds one instance of this class and uses it when necessary or passes
29   * it to other classes that need to use it.
30   *
31   * @author tgayley
32   *
33   */
34  public class ProxySettings {
35  
36      private static ProxySettings instance = null;  // Singleton
37  
38      private int useProxy;  // 0 = false, 1 = automatic, 2 = true
39      private String httpProxyHost;   // For these values, null or 0 means ignore.
40      private int httpProxyPort;
41  
42      private String proxyUsername;
43      private String proxyPassword;
44  
45      public static final int PROXY_NONE = 0;
46      public static final int PROXY_AUTOMATIC = 1;
47      public static final int PROXY_MANUAL = 2;

50      private ProxySettings() {
51          // Try to grab system-specific proxy settings (e.g., from Internet Explorer on Windows).
52          // Actual acquisition of these values is done in setupProxy().
53          System.setProperty("java.net.useSystemProxies", "true");
54          ProxySelector.setDefault(MyProxySelector.getInstance());
55      }

100     // Decide what proxy, if any, to use for a request. Manual settings take precedence,
101     // then we fall back to using system settings (e.g., Internet Explorer settings on Windows)
102     // as handled by the ProxySelector class introduced in Java 1.5.
103     //
104     // Public so that it can be called from Mathematica.
105     //
106     public synchronized String[] getProxyHostAndPort(String url) {
107 
108         int port = 0;
109         String host = null;
110 
111         if (useProxy == PROXY_AUTOMATIC) {
112             ProxySelector ps = ProxySelector.getDefault(); // Will get MyProxySelector.
113             try {
114                 URI uri = new URI(url);
115                 List<Proxy> proxyList = ps.select(uri);
116                 int len = proxyList.size();
117                 for (int i = 0; i < len; i++) {
118                     Proxy p = (Proxy) proxyList.get (i);
119                     InetSocketAddress addr = (InetSocketAddress) p.address();
120                     if (addr != null) {
121                         host = addr.getHostName();
122                         port = addr.getPort();
123                         break;
124                     }
125                 }
126             } catch (Exception e) {
127                 // TODO: Handle?
128             }
129         } else if (useProxy == PROXY_MANUAL) {
130             String protocol;
131             int colonPos = url.indexOf(':');
132             if (colonPos != -1) {
133                 protocol = url.substring(0, colonPos).toLowerCase();
134             } else {
135                 // Shouldn't happen, but might as well do something and let it fail later.
136                 protocol = "http";
137             }
138             if (protocol.equals("http")) {
139                 host = httpProxyHost;
140                 port = httpProxyPort;
141             }
142             // Don't handle directly Socks calls.
143         }
144 
145         return new String[]{host, String.valueOf(port)};
146     }

149     public synchronized HttpHost getProxyForHttpClient(String url) {
150 
151         String[] hostAndPort = getProxyHostAndPort(url);
152         String host = hostAndPort[0];
153         int port = Integer.parseInt(hostAndPort[1]);
154 
155         if (port != 0 && host != null) {
156             return new HttpHost(host, port, "http");
157         } else {
158             return null;
159         }
160     }
161

163     public synchronized Proxy getProxyForJavaNet(String url) {
164 
165         String[] hostAndPort = getProxyHostAndPort(url);
166         String host = hostAndPort[0];
167         int port = Integer.parseInt(hostAndPort[1]);
168 
169         // If we got settings for both host and port, use them; otherwise no proxy.
170         if (port != 0 && host != null)
171             return new Proxy(Proxy.Type.HTTP, new InetSocketAddress(host, port));
172         else
173             return null; // Don't return Proxy.NO_PROXY, as that seems not to be a usable value--just a sentinel.
174     }

177     // This class is plugged in as the default system-wide ProxySelector.
178     // It's a slightly weird setup, since ProxySettings calls this class for Automatic
179     // proxy configuration, and this class calls back to ProxySettings for Manual and
180     // Direct proxy settings. Things would be a little cleaner if we didn't have to worry
181     // about Java 1.4 compatibility (this class inherits from a 1.5-only class).
182 
183     static class MyProxySelector extends ProxySelector {
184 
185         static MyProxySelector instance = null;
186         ProxySelector origSelector = null;
187 
188         // The list we return when we want to indicate to the caller that no proxy should be used.
189         // In older versions of Java you could return an empty list, but in newer ones there is a bug
190         // where the java.net code will throw a NullPointerException if you return an empty list.
191         private final List<Proxy> NO_PROXY_LIST = new ArrayList<Proxy>(1);
192 
193         private MyProxySelector() {
194             origSelector = ProxySelector.getDefault();
195             NO_PROXY_LIST.add(Proxy.NO_PROXY);
196         }
197 
198         static synchronized ProxySelector getInstance() {
199             if (instance == null)
200                 instance = new MyProxySelector();
201             return instance;
202         }
203 
204         public List<Proxy> select(URI uri) {
205 
206             int useProxy = ProxySettings.getInstance().getUseProxy();
207             if (useProxy == PROXY_AUTOMATIC)
208                 return origSelector.select(uri);
209             else if (useProxy == PROXY_MANUAL) {
210                 Proxy p = (Proxy) ProxySettings.getInstance().getProxyForJavaNet(uri.toString());
211                 if (p != null) {
212                     List<Proxy> proxies = new ArrayList<Proxy>(1);
213                     proxies.add(p);
214                     return proxies;
215                 } else {
216                     return NO_PROXY_LIST;
217                 }
218             } else {
219                 return NO_PROXY_LIST;
220             }
221         }
222 
223         public void connectFailed(URI uri, SocketAddress sa, IOException ioe) {
224             origSelector.connectFailed(uri, sa, ioe);
225         }
226 
227     }
