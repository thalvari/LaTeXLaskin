6   package latexlaskin.wacalculator;

8   import com.wolfram.alpha.*;
9   import java.util.*;

11  /**
12   * Tekee laskutoimitukset hyödyntäen API:a.
13   *
14   * @author thalvari
15   */
16  public class WACalculator {
17  
18      // Vastaavat xml-tiedoston rikkinäisiä symboleita.
19      private static final char EQUALS = 63449;
20      private static final char NEPER = 63309;
21      private static final char IMAGINARY = 63310;
22  
23      private static final String[] SUPPORTED_POD_IDS = {
24          "Result", "AlternateForm", "ExpandedForm", "DecimalApproximation",
25          "AlternateFormAssumingAllVariablesAreReal"
26      };
27      private static final String[] SUPPORTED_FIRST_POD_TITLES = {
28          "Derivative", "Indefinite integral", "Definite integral"
29      };
30  
31      private final WAEngine engine;
32      private String error;
33      private boolean debug;
34  
35      /**
36       * Luokan konstruktori.
37       *
38       * @param appid Käyttäjän AppID.
39       * @param format Mitä halutaan mukaan vastaukseen.
40       * @param debug Käytetäänkö debuggaustilaa.
41       */
42      public WACalculator(String appid, String format, boolean debug) {
43          engine = new WAEngine();
44          engine.setAppID(appid);
45          engine.addFormat(format);
46          this.debug = debug;
47      }
48  
49      /**
50       * Palauttaa luodun laskumoottorin.
51       *
52       * @return laskumoottori
53       */
54      public WAEngine getEngine() {
55          return engine;
56      }
57  
58      /**
59       * Palauttaa mahdollisen virheviestin.
60       *
61       * @return Virheviesti.
62       */
63      public String getError() {
64          return error;
65      }
66  
67      /**
68       * Asettaa debug-tilan päälle tai pois.
69       *
70       * @param debug Uusi tila.
71       */
72      public void setDebug(boolean debug) {
73          this.debug = debug;
74      }
75  
76      /**
77       * Kysyy API:lta laskun tulosta.
78       *
79       * @param input Laskutoimitus tekstinä.
80       * @return Lista vaihtoehtoisista ratkaisuista.
81       */
82      public List<String> query(String input) {
83          if (input.isEmpty()) {
84              error = "Syöte virheellinen.";
85              return null;
86          }
87          WAQuery query = createQuery(input);
88          WAQueryResult queryResult;
89          try {
90              queryResult = engine.performQuery(query);
91          } catch (WAException e) {
92              error = "Tapahtui verkkovirhe.";
93              return null;
94          }
95          if (!checkResults(queryResult)) {
96              return null;
97          } else {
98              return processResults(queryResult);
99          }
100     }
101 
102     private WAQuery createQuery(String input) {
103         WAQuery query = engine.createQuery();
104         query.setInput(input);
105         if (debug) {
106             printDebug(query);
107         }
108         return query;
109     }
110 
111     private boolean checkResults(WAQueryResult queryResult) {
112         if (queryResult.isError()) {
113             error = "AppID virheellinen.";
114             return false;
115         } else if (!queryResult.isSuccess()) {
116             error = "Syöte virheellinen.";
117             return false;
118         }
119         return true;
120     }
121 
122     private List<String> processResults(WAQueryResult queryResult) {
123         List<String> results = extractResults(queryResult);
124         results = removeBadResults(results);
125         trimResults(results);
126         replaceSymbols(results);
127         return results;
128     }
129 
130     private List<String> extractResults(WAQueryResult queryResult) {
131         List<String> results = new ArrayList();
132         for (WAPod pod : queryResult.getPods()) {
133             if (isSupported(pod)) {
134                 results.addAll(extractResults(pod));
135             }
136         }
137         return results;
138     }
139 
140     private boolean isSupported(WAPod pod) {
141         return Arrays.asList(SUPPORTED_POD_IDS).contains(pod.getID())
142                 || (Arrays.asList(SUPPORTED_FIRST_POD_TITLES).contains(
143                         pod.getTitle()) && pod.getPosition() == 100);
144     }
145 
146     private List<String> extractResults(WAPod pod) {
147         List<String> results = new ArrayList();
148         for (WASubpod subpod : pod.getSubpods()) {
149             for (Object o : subpod.getContents()) {
150                 results.add(((WAPlainText) o).getText());
151             }
152         }
153         return results;
154     }
155 
156     private List<String> removeBadResults(List<String> results) {
157         List<String> goodResults = new ArrayList();
158         for (String result : results) {
159             if (!result.contains("(for")) {
160                 goodResults.add(result);
161             }
162         }
163         return goodResults;
164     }
165 
166     private void trimResults(List<String> results) {
167         for (int i = 0; i < results.size(); i++) {
168             results.set(i, trimResult(results.get(i)));
169         }
170     }
171 
172     private String trimResult(String result) {
173         int idx = result.indexOf(EQUALS);
174         if (idx != -1) {
175             result = result.substring(idx + 1);
176         }
177         idx = result.indexOf('≈');
178         if (idx != -1) {
179             result = result.substring(0, idx);
180         }
181         return result;
182     }
183 
184     private void replaceSymbols(List<String> results) {
185         for (int i = 0; i < results.size(); i++) {
186             results.set(i, replaceSymbols(results.get(i)));
187         }
188     }
189 
190     private String replaceSymbols(String result) {
191         char[] chars = result.toCharArray();
192         for (int i = 0; i < result.length(); i++) {
193             if (chars[i] == NEPER) {
194                 chars[i] = 'e';
195             } else if (chars[i] == IMAGINARY) {
196                 chars[i] = 'i';
197             }
198         }
199         return new String(chars);
200     }
201 
202     private void printDebug(WAQuery query) {
203         System.out.println("Ratkaisu WA:n sivuilla:");
204         System.out.println(query.toWebsiteURL());
205         System.out.println("Ratkaisu XML-tiedostona:");
206         System.out.println(engine.toURL(query));
207     }
